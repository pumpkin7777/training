#include <stdio.h>


int sub(long long *n, int i) {
    if (*n == 4) {
        s[i] += *n / 2;
        *n = *n / 2;
        return 1;
    }
    if (*n % 2 == 0 && (*n / 2) % 2 == 1) {
        s[i] += *n / 2;
        *n = *n / 2;
    } else {
        s[i]++;
        (*n)--;
    }
    if (*n > 0)
        return sub(n, 3 - i);
    else
        return 0;
}

int main() {
    int t;
    long long s[3];
    scanf("%d", &t);

    while (t--) {
        s[1] = 0, s[2] = 0;
        long long n;
        scanf("%lld", &n);
        while (n > 0) {
            sub(&n, 1);
            if (n > 0)
                sub(&n, 2);
            n = n - 2;
        }
        printf("%lld\n", s[1]);
    }

    return 0;
}
